\noindent \textbf{Exercise 13.1, Q5.} If $\pr{A} = \frac{6}{11}$, $\pr{B} = \frac{5}{11}$ and $\pr{A \cup B} = \frac{7}{11}$, find

\begin{enumerate}[label=(\roman{enumi})]
\item $\pr{A \cap B}$
\item $\pr{A|B}$
\item $\pr{B|A}$
\end{enumerate}

\solution Define the random variables $X$ and $Y$ as follows:

\begin{align}
X = 
\begin{cases}
0, & \text{event A occurs} \\
1, & \text{otherwise} \\
\end{cases} \\
Y = 
\begin{cases}
0, & \text{event B occurs} \\
1, & \text{otherwise} \\
\end{cases}
\label{defn}
\end{align}

We know that, given events A and B,
\begin{align}
\pr{AB} = \pr{A} + \pr{B} - \pr{A + B}
\label{eq:union}
\end{align}
and also, 
\begin{align}
\pr{A|B} = \frac{\pr{AB}}{\pr{B}}
\label{eq:conditional}
\end{align}
Using \eqref{defn} and \eqref{eq:union}, we get 
\begin{align}
\pr{X = 0, Y = 0} = \frac{6}{11} + \frac{5}{11} - \frac{7}{11} = \frac{4}{11}
\label{sol:1}
\end{align}
From \eqref{eq:conditional} and \eqref{sol:1},
\begin{align}
\pr{X = 0|Y = 0} = \frac{\pr{X = 0, Y = 0}}{\pr{Y = 0}} = \frac{4}{5} 
\label{sol:2} 
\end{align}
and
\begin{align}
\pr{X = 0|Y = 0} = \frac{\pr{X = 0, Y = 0}}{\pr{X = 0}}  = \frac{2}{3}
\label{sol:3}
\end{align}

\noindent The C code \texttt{./codes/7{\_}1.c} verifies the solution, within limits of float precision.

\noindent \textbf{Note:} Derivation of \eqref{eq:union} using Boolean Algebra:
\begin{align}
LHS = A \vee B &= \neg(\neg A \wedge \neg B) \\
&= 1 - (1 - A)(1 - B) \\
&= A + B - AB = RHS
\label{derive}
\end{align}
The following identities were used:
\begin{align}
\text{(De-Morgan's Law) }&\neg(A \vee B) = \neg A \wedge \neg B \\
&\neg A = 1 - A \\
&A \wedge B = AB 
\end{align}
